756 inline bool Clock::isUpdate(const ClockID _id)
757 {
758 std::shared_ptr<ClockElem> i(nullptr);
759 try
760 {
761 i = getIterator(_id);
762 if (i.get() == nullptr)
763 throw ClockNotFoundException();
764
765 if (i->pause == true)
766 {
767 return false;
768 }
769 std::lock_guard<std::mutex> lock(i->templock);
770 QueryPerformanceCounter(&i->temp_lint);
771 if (i->temp_lint.QuadPart - i->cycle > i->update_tick)
772 {
773 i->last_cycle = i->cycle;
774 i->cycle = i->temp_lint.QuadPart;
775 auto relative_passed = (i->cycle - i->last_cycle) * i->scale;
776 i->relative_tick += relative_passed;
777#ifdef EVENT_SYSTEM
778 EventSystem& eventSystem = EventSystem::Instance();
779 QueryPerformanceFrequency(&i->temp_lint);
780 auto freq = i->temp_lint.QuadPart;
781 for (auto& e : i->eventList)
782 {
783 eventSystem.TriggerEventUpdate(*e, relative_passed * 1000 / freq);
784 }
785#endif // EVENT_SYSTEM
786 return true;
787 }
788 else
789 {
790 return false;
791 }
792 }
793 catch (ClockException& exp)
794 {
795 std::cout << "\n::Clock::isUpdate()" << exp.what() << std::endl;
796 }
797 return false;
