780 inline bool Clock::isUpdate(const ClockID _id)
781 {
782 std::shared_ptr<ClockElem> i(nullptr);
783 try
784 {
785 i = getIterator(_id);
786 if (i.get() == nullptr)
787 throw ClockNotFoundException();
788
789 if (i->pause == true)
790 {
791 return false;
792 }
793 std::lock_guard<std::mutex> lock(i->templock);
794 QueryPerformanceCounter(&i->temp_lint);
795 if (i->temp_lint.QuadPart - i->cycle > i->update_tick)
796 {
797 i->last_cycle = i->cycle;
798 i->cycle = i->temp_lint.QuadPart;
799 auto relative_passed = (i->cycle - i->last_cycle) * i->scale;
800 i->relative_tick += relative_passed;
801#ifdef EVENT_SYSTEM
802 EventSystem& eventSystem = EventSystem::Instance();
803 QueryPerformanceFrequency(&i->temp_lint);
804 auto freq = i->temp_lint.QuadPart;
805 std::lock_guard<std::mutex> lock(i->lock);
806 for (auto e : i->eventList)
807 {
808 eventSystem.TriggerEventUpdate(e, relative_passed * 1000 / freq);
809 }
810#endif // EVENT_SYSTEM
811 return true;
812 }
813 else
814 {
815 return false;
816 }
817 }
818 catch (ClockException& exp)
819 {
820 std::cout << "\n::Clock::isUpdate()" << exp.what() << std::endl;
821 }
822 return false;
